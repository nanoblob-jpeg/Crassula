#version 330 core
layout (location = 0) in vec4 vertex;
layout (location = 1) in vec2 aOffset;

uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;

out vec2 TexCoords;

void main()
{
    gl_Position = projection * view * model * vec4(vertex.xy + aOffset, 0.0, 1.0);
    TexCoords = vertex.zw;
}
